Write better code!

Real World Hazzle

  • Huge classes/files/packages
  • Difficult to read
  • Tight coupling
  • Low cohesion
  • ... and all that other stuff

Nine Rules

  1. One level of indentation per method
  2. Don't use the ELSE keyword
  3. Wrap all primitives and Strings
  4. First class collections
  5. One dot per line
  6. Don't abbreviate
  7. Keep all entities small
  8. No classes with more than two instance variables
  9. No getters, setters or properties

1

One level of indentation per method

class Board {
    private int[][] data = /* ... */;

    public String toString() {
        StringBuffer buffer = new StringBuffer();
        for (int i = 0; i < 10; i++) {
            for (int j = 0; j < 10; j++) {
                buffer.append(data[i][j]);
                buffer.append("\n");
            }
        }
    }
}

class Board {
    private int[][] data = /* ... */;

    public String toString() {
        StringBuffer buffer = new StringBuffer();
        collectRows(buffer);
        return buffer.toString();
    }

    private collectRows(StringBuffer buffer) {
        for (int i = 0; i < 10; i++) {
            collectRow(buffer, i);
        }
    }

    private collectRow(StringBuffer buffer, int row) {
        for (int j = 0; j < 10; j++) {
            buffer.append(data[row][j]);
        }
        buffer.append("\n");
    }
}
2

Don't use the ELSE keyword

status = service.someOperation();
if (status == DONE) {
    doSomething();
} else if (status == NOT_FOUND && time == today) {
    doSomethingUrgent();
} else if (status == NOT_FOUND) {
    doSomethingElse();
} else {
    reportError();
}
Performable action = service.someOperation();
action.perform();

interface Performable {
    void perform();
}

class NotFoundAction implements Performable {
    public void perform() {
        // Do stuff.
    }
}

5

One dot per line

class Piece {
    String representation;
}

class Location {
    Piece current;
}

class Board {
    String boardRepresentation() {
        StringBuffer buffer = new StringBuffer();
        for (Location location : getSquares()) {
            buffer.append(location.current.representation);
        }
        return buffer.toString();
    }
}

class Piece {
    String representation;
    void addTo(StringBuffer buffer) {
        buffer.append(representation);
    }
}

class Location {
    Piece current;
    void addTo(StringBuffer buffer) {
        current.addTo(buffer);
    }
}

class Board {
    String boardRepresentation() {
        StringBuffer buffer = new StringBuffer();
        for (Location location : getSquares()) {
            location.addTo(buffer);
        }
        return buffer.toString();
    }
}
