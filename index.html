Write better code!

Real World Hazzle

  • Huge classes/files/packages
  • Difficult to read
  • Tight coupling
  • Low cohesion
  • ... and all that other stuff

Nine Rules

  1. One level of indentation per method
  2. Don't use the ELSE keyword
  3. Wrap all primitives and Strings
  4. First class collections
  5. One dot per line
  6. Don't abbreviate
  7. Keep all entities small
  8. No classes with more than two instance variables
  9. No getters, setters or properties

1

One level of indentation per method

class Board {
    private int[][] data = /* ... */;

    public String toString() {
        StringBuffer buffer = new StringBuffer();
        for (int i = 0; i < 10; i++) {
            for (int j = 0; j < 10; j++) {
                buffer.append(data[i][j]);
                buffer.append("\n");
            }
        }
        return buffer.toString();
    }
}

class Board {
    private int[][] data = /* ... */;

    public String toString() {
        StringBuffer buffer = new StringBuffer();
        collectRows(buffer);
        return buffer.toString();
    }

    private void collectRows(StringBuffer buffer) {
        for (int i = 0; i < 10; i++) {
            collectRow(buffer, i);
        }
    }

    private void collectRow(StringBuffer buffer, int row) {
        for (int j = 0; j < 10; j++) {
            buffer.append(data[row][j]);
        }
        buffer.append("\n");
    }
}

Don't use the ELSE keyword

    doSomethingElse();

    reportError();
}

    void perform();

5

One dot per line

class Piece {
    public String representation;
}

class Location {
    public Piece piece;
}

class Board {
    public String boardRepresentation() {
        StringBuffer buffer = new StringBuffer();
        for (Location location : getLocations()) {
            buffer.append(location.piece.representation);
        }
        return buffer.toString();
    }
}

class Piece {
    private String representation;
    public void addTo(StringBuffer buffer) {
        buffer.append(representation);
    }
}

class Location {
    private Piece piece;
    public void addTo(StringBuffer buffer) {
        piece.addTo(buffer);
    }
}

class Board {
    public String boardRepresentation() {
        StringBuffer buffer = new StringBuffer();
        for (Location location : getLocations()) {
            location.addTo(buffer);
        }
        return buffer.toString();
    }
}
